214 static void invoke_callbacks(Tundra::KEYCODE key)
215 {
216 // Invalid key.
217 if(static_cast<uint8_t>(key) > 127) return;
218
219 const std::vector<ICallback*>& targ_vec
220 = s_keys_to_callback_funcs.at(static_cast<uint8_t>(key));
221
222 for(ICallback* callback : targ_vec)
223 {
224 // tracker.function(tracker.obj_instance);
225 callback->invoke();
226 }
227 }

235 static void handle_pressed_key_check(Tundra::KEYCODE key)
236 {
237 uint64_t& blocked_timestamp = s_keys_to_blocked_timestamps[
238 static_cast<uint8_t>(key)];
239
240 // If this key is blocked
241 if(TimeObserver::get_elapsed() <= blocked_timestamp) return;
242
243 // Key is not blocked, set the blocked timestamp to 0 and invoke
244 // callbacks.
245 blocked_timestamp = 0;
246
247 invoke_callbacks(key);
248 }
